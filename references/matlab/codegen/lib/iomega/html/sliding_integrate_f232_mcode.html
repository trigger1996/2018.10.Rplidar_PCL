1function [ dst_x, dst_y, vel_x, vel_y ] = sliding_integrate_f(ax_in, ay_in, ts, arr_len_in)
  2
  3% 核心思想：
  4% 记录加速度序列，每次以滑动的方式添加一个加速度进来，积分得到速度，这样就可以利用频域，又能单数据更新
  5% 但是这样速度和位移的积分结果就只和加速度序列有关，而与全局状态无关，这显然是错的——比如如果停止运动，过一段时间加速度会很小，然后速度和位移也会趋于0
  6% 所以需要让这些参数都和全局有关起来
  7% 所以同样地记录速度即可
  8
  9global acc_x_arr;                                       % persistent
 10global acc_y_arr;                        
 11global vel_x_arr;
 12global vel_y_arr;
 13global arr_len;
 14global t;
 15
 16% if size(arr_len_in, 1) == 0
 17%     arr_len = [];                                       % 预留用来复位的，只要arr_len_in输入等于[]，就复位
 18%     vel_x_arr = []; vel_y_arr = [];                     % 编译的时候记得注释掉这个，用不上，还会造成编译错误
 19%     acc_x_arr = []; acc_y_arr = [];
 20%     t         = [];
 21%     dst_x = 0;      dst_y = 0;
 22%     vel_x = 0;      vel_y = 0;
 23%     return;
 24% end
 25
 26if size(arr_len, 1) == 0                                % 是否第一次调用
 27    arr_len = arr_len_in;
 28end
 29
 30if size(acc_x_arr, 1) == 0 || size(acc_y_arr, 1) == 0	% 如果是第一次调用，则给滑动数列赋空值
 31    acc_x_arr = zeros(arr_len, 1);	acc_y_arr = zeros(arr_len, 1);
 32    vel_x_arr = zeros(arr_len, 1);	vel_y_arr = zeros(arr_len, 1);
 33    t = (0 : ts : (arr_len - 1) * ts)';
 34end
 35
 36% 如果数据不满，则不进行运算
 37if acc_x_arr(size(acc_x_arr, 1), :) == 0. || acc_y_arr(size(acc_y_arr, 1), :) == 0.
 38    acc_x_arr(min(find(acc_x_arr == 0))) = ax_in;       % 找到最后一个0（未填满）的位置插入数据
 39    acc_y_arr(min(find(acc_y_arr == 0))) = ay_in;
 40    dst_x = 0;      dst_y = 0;
 41    vel_x = 0;      vel_y = 0;
 42    return;
 43end
 44    
 45% 正常计算
 46% 将数据滑动前移一位
 47for i = 1 : arr_len - 1
 48    acc_x_arr(i, :) = acc_x_arr(i + 1, :);
 49    acc_y_arr(i, :) = acc_y_arr(i + 1, :);
 50end
 51% 最后一位插入新的数据
 52    acc_x_arr(arr_len, :) = ax_in;
 53    acc_y_arr(arr_len, :) = ay_in;    
 54    
 55% 如果vel序列没有填满则记录vel序列后退出
 56% 此时加速度已经能计算，那么先根据加速度序列算出速度序列，然后推出
 57if vel_x_arr(size(vel_x_arr, 1), :) == 0. || vel_y_arr(size(vel_y_arr, 1), :) == 0.
 58    vel_x_arr =  iomega(acc_x_arr, ts, 3, 2);
 59    vel_y_arr =  iomega(acc_y_arr, ts, 3, 2);    
 60    dst_x = 0;      dst_y = 0;
 61    vel_x = 0;      vel_y = 0;
 62    return;
 63end
 64
 65
 66    
 67% 积分
 68    % 频域积分
 69    % 速度
 70    vel_x_int =  iomega(acc_x_arr, ts, 3, 2);
 71    vel_x_int =  detrend(vel_x_int);
 72    vel_y_int =  iomega(acc_y_arr, ts, 3, 2);
 73    vel_y_int =  detrend(vel_y_int);    
 74    
 75    % 将数据滑动前移一位
 76    for i = 1 : arr_len - 1
 77        vel_x_arr(i, :) = vel_x_arr(i + 1, :);
 78        vel_y_arr(i, :) = vel_y_arr(i + 1, :);        
 79    end
 80    % 最后一位插入新的数据
 81        vel_x_arr(arr_len, :) = vel_x_int(size(vel_x_int, 1), :);
 82        vel_y_arr(arr_len, :) = vel_y_int(size(vel_y_int, 1), :);    
 83        
 84    %disint =  iomega(acc_arr, ts, 3, 1);
 85    dis_x_int =  iomega(vel_x_arr, ts, 3, 2);
 86    % 去除位移中的二次项
 87    px = polyfit(t, dis_x_int, 2);
 88    dis_x_int = dis_x_int - polyval(px, t);
 89    
 90    %disint =  iomega(acc_arr, ts, 3, 1);
 91    dis_y_int =  iomega(vel_y_arr, ts, 3, 2);
 92    % 去除位移中的二次项
 93    py = polyfit(t, dis_y_int, 2);
 94    dis_y_int = dis_y_int - polyval(py, t);
 95    
 96
 97dst_x = dis_x_int(arr_len, :);      dst_y = dis_y_int(arr_len, :);
 98vel_x = vel_x_arr(arr_len, :);      vel_y = vel_y_arr(arr_len, :);
 99end
