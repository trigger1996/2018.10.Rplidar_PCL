  579               ai = fmod(absa, 4.0);
  580               if (ai == 0.0) {
  581                 ai = 0.0;
  582               } else {
  583                 if (absa < 0.0) {
  584                   ai += 4.0;
  585                 }
  586               }
  587             }
  588           } else {
  589             ai = rtNaN;
  590           }
  591   
  592           ndbl = rt_roundd_snf(ai);
  593           if (ndbl < 128.0) {
  594             r = (signed char)ndbl;
  595           } else {
  596             r = 0;
  597           }
  598   
  599           if (r == 3) {
  600             ndbl = 0.0;
  601             cdiff = -cdiff;
  602           } else if (r == 2) {
  603             ndbl = -cdiff;
  604             cdiff = 0.0;
  605           } else if (r == 1) {
  606             ndbl = 0.0;
  607           } else {
  608             ndbl = cdiff;
  609             cdiff = 0.0;
  610           }
  611         } else if ((ai == 0.0) && rtIsInf(absa) && (fabs(ndbl) == 1.0)) {
  612           ndbl = 1.0;
  613           cdiff = 0.0;
  614         } else {
  615           if (iomega_array->data[nm1d2].im == 0.0) {
  616             if (iomega_array->data[nm1d2].re < 0.0) {
  617               ndbl = log(fabs(iomega_array->data[nm1d2].re));
  618               cdiff = 3.1415926535897931;
  619             } else {
  620               ndbl = log(fabs(iomega_array->data[nm1d2].re));
  621               cdiff = 0.0;
  622             }
  623           } else if ((fabs(iomega_array->data[nm1d2].re) > 8.9884656743115785E+307)
  624                      || (fabs(iomega_array->data[nm1d2].im) >
  625                          8.9884656743115785E+307)) {
  626             ndbl = log(rt_hypotd_snf(iomega_array->data[nm1d2].re / 2.0,
  627                         iomega_array->data[nm1d2].im / 2.0)) + 0.69314718055994529;
  628             cdiff = rt_atan2d_snf(iomega_array->data[nm1d2].im, iomega_array->
  629                                   data[nm1d2].re);
  630           } else {
  631             ndbl = log(rt_hypotd_snf(iomega_array->data[nm1d2].re,
  632                         iomega_array->data[nm1d2].im));
  633             cdiff = rt_atan2d_snf(iomega_array->data[nm1d2].im, iomega_array->
  634                                   data[nm1d2].re);
  635           }
  636   
  637           ndbl *= absa;
  638           cdiff *= absa;
  639           if (cdiff == 0.0) {
  640             ndbl = exp(ndbl);
  641             cdiff = 0.0;
  642           } else if (rtIsInf(cdiff) && rtIsInf(ndbl) && (ndbl < 0.0)) {
  643             ndbl = 0.0;
  644             cdiff = 0.0;
  645           } else {
  646             ai = exp(ndbl / 2.0);
  647             ndbl = ai * (ai * cos(cdiff));
  648             cdiff = ai * (ai * sin(cdiff));
  649           }
  650         }
  651   
  652         ai = A[nm1d2].re;
  653         A[nm1d2].re = A[nm1d2].re * ndbl - A[nm1d2].im * cdiff;
  654         A[nm1d2].im = ai * cdiff + A[nm1d2].im * ndbl;
  655       } else {
  656         A[nm1d2].re = 0.0;
  657         A[nm1d2].im = 0.0;
  658       }
  659     }
  660   
  661     emxFree_creal_T(&iomega_array);
  662   
  663     /*    Shift new frequency-amplitude array back to MATLAB format and */
  664     /*    transform back into the time domain via the inverse FFT. */
  665     ifftshift(A);
  666   
  667     /*    Remove zeros that were added to datain in order to pad to next */
  668     /*    biggerst power of 2 and return dataout. */
  669     ifft(A, dcv0);
  670     for (nm1d2 = 0; nm1d2 < 512; nm1d2++) {
  671       dataout[nm1d2] = dcv0[nm1d2].re;
  672     }
  673   }
