1function [ dst, vel ] = sliding_integrate_f(data_in, ts, arr_len_in)
 2
 3% 核心思想：
 4% 记录加速度序列，每次以滑动的方式添加一个加速度进来，积分得到速度，这样就可以利用频域，又能单数据更新
 5% 但是这样速度和位移的积分结果就只和加速度序列有关，而与全局状态无关，这显然是错的——比如如果停止运动，过一段时间加速度会很小，然后速度和位移也会趋于0
 6% 所以需要让这些参数都和全局有关起来
 7% 所以同样地记录速度即可
 8
 9global acc_arr;                     % persistent
10global vel_arr;
11global arr_len;
12global t;
13
14% if size(arr_len_in, 1) == 0
15%     arr_len = [];                   % 预留用来复位的，只要arr_len_in输入等于[]，就复位
16%     vel_arr = [];
17%     acc_arr = [];
18%     t       = [];
19%     dst = 0;
20%     vel = 0;
21%     return;
22% end
23
24if size(arr_len, 1) == 0            % 是否第一次调用
25    arr_len = arr_len_in;
26end
27
28if size(acc_arr, 1) == 0            % 如果是第一次调用，则给滑动数列赋空值
29    acc_arr = zeros(arr_len, 1);
30    vel_arr = zeros(arr_len, 1);
31    t = (0 : ts : (arr_len - 1) * ts)';
32end
33
34% 如果数据不满，则不进行运算
35if acc_arr(size(acc_arr, 1), :) == 0.
36    acc_arr(min(find(acc_arr == 0))) = data_in; % 找到最后一个0（未填满）的位置插入数据
37    dst = 0; vel = 0;
38    return;
39end
40    
41% 正常计算
42% 将数据滑动前移一位
43for i = 1 : arr_len - 1
44    acc_arr(i, :) = acc_arr(i + 1, :);
45end
46% 最后一位插入新的数据
47    acc_arr(arr_len, :) = data_in;
48    
49% 如果vel序列没有填满则记录vel序列后退出
50% 此时加速度已经能计算，那么先根据加速度序列算出速度序列，然后推出
51if vel_arr(size(acc_arr, 1), :) == 0.
52    vel_arr =  iomega(acc_arr, ts, 3, 2);
53    dst = 0; vel = 0;
54    return;
55end
56
57
58    
59% 积分
60    % 频域积分
61    % 速度
62    velint =  iomega(acc_arr, ts, 3, 2);
63    velint =  detrend(velint);
64    
65    % 将数据滑动前移一位
66    for i = 1 : arr_len - 1
67        vel_arr(i, :) = vel_arr(i + 1, :);
68    end
69    % 最后一位插入新的数据
70        vel_arr(arr_len, :) = velint(size(velint, 1), :);
71    
72    %disint =  iomega(acc_arr, ts, 3, 1);
73    disint =  iomega(vel_arr, ts, 3, 2);
74    % 去除位移中的二次项
75    p = polyfit(t, disint, 2);
76    disint = disint - polyval(p, t);
77
78 vel = vel_arr(arr_len, :);
79 dst = disint(arr_len,  :);
80 
81end
